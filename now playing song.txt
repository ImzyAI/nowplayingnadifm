<script>
  // Function untuk update waktu masa sebenar
  function updateTime() {
    const now = new Date();
    const timeString = now.toLocaleTimeString();
    document.getElementById('current-time').textContent = timeString;
  }

  setInterval(updateTime, 1000);

  // Update title lagu dan nama artis (guna data dari radio.co)
  document.addEventListener('DOMContentLoaded', () => {
    fetch('https://embed.radio.co/embed/s9f71ec2d1/song.js')
      .then(response => response.json())
      .then(data => {
        document.getElementById('song-title').textContent = data.song.title;
        document.getElementById('artist-name').textContent = data.song.artist;
      });

    fetch('https://embed.radio.co/embed/s9f71ec2d1/artwork.js')
      .then(response => response.json())
      .then(data => {
        document.getElementById('album-art').src = data.song.artwork_url;
      });
  });
</script>
